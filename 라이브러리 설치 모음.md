# 📜 라이브러리 설치 명령어 모음

아래 코드를 터미널에 한번에 입력하면 순차적으로 설치됩니다.

```bash
# HTTP 클라이언트
# Axios
npm i axios

# HTML 클래스 유틸
# tailwind-merge
npm i tailwind-merge
# clsx
npm i clsx
# cva
npm i class-variance-authority

# 서버 상태 관리
# React Query
npm i @tanstack/react-query
# React Query Dev tool
npm install @tanstack/react-query-devtools

# 클라이언트 상태 관리
# zustand
npm i zustand

# 유효성 관리
# zod
npm i zod
# React-hook-form
npm i react-hook-form
# @hookform-resolvers
npm i @hookform/resolvers

# 디자인/UI
# React icons
npm i react-icons
npm i -D @types/react-icons
# framer-motion
npm i framer-motion
```
